FIS 使用心得五（解决方案封装）

这一章，介绍一下 FIS的解决方案封装

其实对于解决方案的封装，最好的方法是下一个官方的fis-pure（百度官方的一个纯前端解决方案）读它的代码，代码并不多。

捆绑插件

一个解决方案，必然依赖很多插件，为了不让每次使用这个解决方案的时候都一个个的去安装必须的插件，我们可以将所有的插件保存在解决方案的依赖中。这样，你只要全局安装你的解决方案，其他依赖插件就会自动安装。至于怎么添加到依赖中，我相信用过npm的童鞋都熟悉了。

// 就是在根目录安装插件是 加上'--save'参数
npm install fis-parser-marked --save

覆写插件

当fis调用一个需要的插件的时候，一般是查找’fis-‘开头的插件，但是有时候，你自己的解决方案，你可能希望用自己的插件覆盖fis的插件。这是你可以执行如下配置：

fis.require.prefixes.unshift('hello');

这样fis就会优先执行’hello-‘开头的插件

方案封装

所谓解决方案，其实就是按照自己项目的需要，将需要的插件和相应的插件配置，还有产入产出等等一些列配置击中起来，放在一起，而不必在每个项目都去重复的配置。

公用的配置放在解决方案的’包’的主文件中。而各个项目特别的配置放在项目的‘fis-conf.js’中

解决方案示例

下面是我封装的一个自己用的解决方案‘sherry’，使用stylus、markdown、requirjs

  • 决绝方案主文件sherry.js

    var fis = module.exports = require('fis');

fis.cli.name = 'sherry';
fis.cli.info = fis.util.readJSON(__dirname + '/package.json');
fis.require.prefixes.unshift('sherry');

fis.config.merge({
    'res': '/res',
    'tpl': '/tpl',
    'view': '',
    project : {
        fileType : {
            text : 'tpl, styl, ejs, md',
            image : 'swf, cur, ico'
        }
    },
    modules: {
        parser: {
            styl: 'stylus',
            md: 'marked'
        },
        preprocessor: {
            html: ['replace', 'uri'],
            js: ['replace'],
            css: ['replace']
        },
        postprocessor: {
            js: ['jswrapper', 'require-async'],
            html: ['require-async']
        },
        postpackager : ['autoload', 'simple'],
        spriter: ['csssprites'],
        lint : {
            js : 'jshint'
        }
    },
    roadmap: {
        ext: {
            styl: 'css',
            ejs: 'html',
            tpl: 'html',
            md: 'html'
        },
        path : [

            {
                reg : /^\/view\/(.*)$/i,
                useCache : false,
                release : '${view}/$1'
            },

            {
                reg : /^\/tpl\/(.*)$/i,
                useCache : false,
                release : '${tpl}/$1'
            },

            {
                reg : /^\/res\/(.*$)/i,
                isMod : true,
                release : '${res}/$1'
            },

            // map json
            {
                reg: /^.*map\.json$/,
                release: '/${res}/pkg/map'
            },

            // 打包文件
            {
                reg: /^\/pkg\/(.*)$/i,
                release: '/${res}/pkg/$1'
            },

            // 其余的
            {
                reg: /^\/.*\.(md|bat)$/,
                release: false
            },
            {
                reg: '**',
                release: '$&'
            }
        ]
    },
    settings: {
        parser: {
            ejs: {
                open : ',
		                close : '
            }
        },
        postprocessor: {
            jswrapper: {
                type: 'amd',
                template: 'define(function(require, exports, module) {${content}});'
            }
        },
        postpackager: {
            autoload: {
                useSiteMap: true,
                useInlineMap: true,
                subpath: 'pkg/map.js'
            },
            simple: {
                autoCombine: true,
                autoReflow: true
            }
        },
        lint : {
            jshint : {
                camelcase : true,
                curly : true,
                eqeqeq : true,
                forin : true,
                immed : true,
                latedef : true,
                newcap : true,
                noarg : true,
                noempty : true,
                node : true,
                ignored: ['lib/**']
            }
        },
        spriter: {
            csssprites: {
                margin: 20
            }
        }
    }
});

  • 具体实用项目的配置文件fis-conf.js

    // ======================= 产出目录 ===============================
fis.config.merge({
    'res': '/res',
    'tpl': '/tpl',
    'view': '/view',
    deploy: {
        local: [{
            to: '../'
        }]
    }
});

// ==== 静态资源域名，使用release命令时，添加--domains或-D参数即可生效
fis.config.set('roadmap.domain', 'http://dev.myweb.com/season/output');

// sherry-preprocessor-replace变量替换
fis.config.set('settings.preprocessor.replace', {
    vars: [
        {
            from: /__ROOT__/g,
            to: 'http://dev.myweb.com/season/output'
        }
    ]
});

// markdown文件不做代码高亮
fis.config.set('settings.parser.marked.highlight', function(code, lang) {
    return code;
});

// 定义不需要模块化封装的第三方库
fis.config.get('roadmap.path').unshift({   
    reg: /^\/res\/(lib\/.*)$/i,
    isMod: false,
    release: '/${res}/$1'
});

// ================ 开启css sprite ===================
// 为所有样式资源开启csssprites
fis.config.get('roadmap.path').unshift({   
    reg: /^\/res\/(.*)\.(css|styl|less)$/i,
    useSprite: true,
    release: '/${res}/$1.css'
});

// ================ 开启 amd ===================
// 配置amd
fis.config.set('modules.postprocessor.tpl', 'jswrapper, amd');
fis.config.set('modules.postprocessor.html', 'jswrapper, amd');
fis.config.set('modules.postprocessor.js', 'jswrapper, amd');
fis.config.set('settings.postprocessor.amd', {
    baseUrl: './res/',
    paths: {
        // 相对于  baseUrl
        // 如果是绝对路径则相对与 root.
        // base 的值可以是字符串，也可以是数组，会按顺序超找。
        // base: './base/base.js'
    },
    packages: [
        {
            name: 'jquery',
            location: 'lib/jquery',
            main: 'jquery-1.11.2'
        },
        {
            name: 'webuploader',
            location: 'lib/webuploader',
            main: 'webuploader'
        }
    ],

    shim: {
        // 'modules/libs/bootstrap/js/bootstrap.js': ['jquery']
    }

});

// ================ csssprites配置 ===================
// 如果想配置html中的css片段进行csssprites处理，可以像下面这样配置
// fis.config.set('settings.spriter.csssprites', {
//   //开启模板内联css处理,默认关闭
//   htmlUseSprite: true,
//   //默认针对html原生<style></style>标签内的内容处理。
//   //用户可以通过配置styleTag来扩展要识别的css片段
//   //以下是默认<style></style>标签的匹配正则
//   styleReg: /(<style(?:(?=\s)[\s\S]*?["'\s\w\/\-]>|>))([\s\S]*?)(<\/style\s*>|$)/ig

//   //**styleReg规则**
//   //1. 默认不配置styleReg，仅支持html中默认style标签中的css内容
//   //2. 配置styleReg时候，仅支持styleReg匹配到的内容。
//   //3. styleReg正则必须捕获三个分组，
//   //     $1为：开始标签（start tag），
//   //     $2为：内容(content) ,
//   //     $3为：结束标签(end tag)

// ================ 打包 ===================
// 使用 depscombine 是因为，在配置 pack 的时候，命中的文件其依赖也会打包进来。
fis.config.set('modules.packager', 'depscombine');

// ================ fis-postpackger-autoload配置 ===================
fis.config.set('settings.postpackager.autoload.type', 'requirejs');
// useSiteMap设置使用整站/页面异步资源表配置，默认为true
fis.config.set('settings.postpackager.autoload.useSiteMap', false);

  • 实用项目的目录结构示例

    // 产出前
web
 |------index.php
 |------front
          |-----fis-conf.js
         |-----view   // 静态html文件，纯前端网站时使用
                |------include
                |------index.html

         |-----tpl    // 后端模板，如smarty模板等
         |-----res    // 静态资源文件，js、img、css等……
                |------config // 一些配置信息
                |------lib    // jquery等第三方库，不进行amd处理
                |------mod    // 模块，需要进行模块化封装
                |------style  // 样式文件

 产出后
web
 |------index.php
 |------tpl
 |------res
          |------config // 一些配置信息
         |------lib    // jquery等第三方库，不进行amd处理
         |------mod    // 模块，需要进行模块化封装
         |------style  // 样式文件

 |------view
         |------include
         |------index.html

 |------front
web
